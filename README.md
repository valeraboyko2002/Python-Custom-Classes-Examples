### 1. TypedList - –¢–∏–ø–∏–∑–∏—Ä–æ–≤–∞–Ω–Ω—ã–π —Å–ø–∏—Å–æ–∫

–ù–∞—Å–ª–µ–¥—É–µ—Ç—Å—è –æ—Ç –≤—Å—Ç—Ä–æ–µ–Ω–Ω–æ–≥–æ –∫–ª–∞—Å—Å–∞ `list` –∏ –≥–∞—Ä–∞–Ω—Ç–∏—Ä—É–µ—Ç, —á—Ç–æ –≤—Å–µ —ç–ª–µ–º–µ–Ω—Ç—ã –∏–º–µ—é—Ç —É–∫–∞–∑–∞–Ω–Ω—ã–π —Ç–∏–ø.

```python
class TypedList(list):
    def __init__(self, element_type=int):
        self.element_type = element_type
        super().__init__()
    
    def append(self, item):
        if isinstance(item, self.element_type):
            super().append(item)
        else:
            raise TypeError("–û—à–∏–±–∫–∞ —Ç–∏–ø–∞")
```

### 2. My_List - –°–ø–∏—Å–æ–∫ —Å –º–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–∏–º–∏ –æ–ø–µ—Ä–∞—Ü–∏—è–º–∏

–†–∞—Å—à–∏—Ä—è–µ—Ç —Å—Ç–∞–Ω–¥–∞—Ä—Ç–Ω—ã–π —Å–ø–∏—Å–æ–∫ –º–µ—Ç–æ–¥–∞–º–∏ –¥–ª—è —Ä–∞–±–æ—Ç—ã —Å —á–∏—Å–ª–∞–º–∏.

```python
class My_List(list):
    def sum_odd(self):
        """–°—É–º–º–∞ –Ω–µ—á–µ—Ç–Ω—ã—Ö —á–∏—Å–µ–ª –≤ —Å–ø–∏—Å–∫–µ"""
        return sum([x for x in self if x % 2 == 1])
    
    def sum_even(self):
        """–°—É–º–º–∞ —á–µ—Ç–Ω—ã—Ö —á–∏—Å–µ–ª –≤ —Å–ø–∏—Å–∫–µ"""
        return sum([x for x in self if x % 2 == 0])
```

### 3. –°–∏—Å—Ç–µ–º–∞ –∫–ª–∞—Å—Å–æ–≤ —Å –º–Ω–æ–∂–µ—Å—Ç–≤–µ–Ω–Ω—ã–º –Ω–∞—Å–ª–µ–¥–æ–≤–∞–Ω–∏–µ–º

–î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏—è –º–∏–∫—Å–∏–Ω–æ–≤ –∏ –º–Ω–æ–∂–µ—Å—Ç–≤–µ–Ω–Ω–æ–≥–æ –Ω–∞—Å–ª–µ–¥–æ–≤–∞–Ω–∏—è.

```python
    """–ú–∏–∫—Å–∏–Ω –¥–ª—è –¥–æ–±–∞–≤–ª–µ–Ω–∏—è email-—Ñ—É–Ω–∫—Ü–∏–æ–Ω–∞–ª—å–Ω–æ—Å—Ç–∏"""
```

### 4. BankAccount - –ë–∞–Ω–∫–æ–≤—Å–∫–∏–π —Å—á–µ—Ç

–ü–æ–ª–Ω–∞—è —Ä–µ–∞–ª–∏–∑–∞—Ü–∏—è –±–∞–Ω–∫–æ–≤—Å–∫–æ–≥–æ —Å—á–µ—Ç–∞ —Å –∏–Ω–∫–∞–ø—Å—É–ª—è—Ü–∏–µ–π –¥–∞–Ω–Ω—ã—Ö.

```python
class BankAccount:
    """–ö–ª–∞—Å—Å –±–∞–Ω–∫–æ–≤—Å–∫–æ–≥–æ —Å—á–µ—Ç–∞ —Å –∑–∞—â–∏—Ç–æ–π –¥–∞–Ω–Ω—ã—Ö"""
    def __init__(self, account_holder, initial_balance=0):
        self.account_holder = account_holder
        self.__balance = initial_balance
        self.__transactions = []
    
    def deposit(self, amount):
        """–ü–æ–ø–æ–ª–Ω–µ–Ω–∏–µ —Å—á–µ—Ç–∞"""
        if amount <= 0:
            raise ValueError("–°—É–º–º–∞ –¥–æ–ª–∂–Ω–∞ –±—ã—Ç—å –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω–æ–π")
        self.__balance += amount
        self.__transactions.append(f"–ü–æ–ø–æ–ª–Ω–µ–Ω–∏–µ: +{amount}")
        return f"–ü–æ–ø–æ–ª–Ω–µ–Ω–æ {amount}. –ë–∞–ª–∞–Ω—Å: {self.__balance}"
    
    def withdraw(self, amount):
        """–°–Ω—è—Ç–∏–µ —Å–æ —Å—á–µ—Ç–∞"""
        if amount <= 0:
            raise ValueError("–°—É–º–º–∞ –¥–æ–ª–∂–Ω–∞ –±—ã—Ç—å –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω–æ–π")
        if amount > self.__balance:
            raise ValueError("–ù–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ —Å—Ä–µ–¥—Å—Ç–≤")
        self.__balance -= amount
        self.__transactions.append(f"–°–Ω—è—Ç–∏–µ: -{amount}")
        return f"–°–Ω—è—Ç–æ {amount}. –ë–∞–ª–∞–Ω—Å: {self.__balance}"
    
    @property
    def balance(self):
        """–¢–µ–∫—É—â–∏–π –±–∞–ª–∞–Ω—Å (—Ç–æ–ª—å–∫–æ —á—Ç–µ–Ω–∏–µ)"""
        return self.__balance
    
    def get_transactions(self):
        """–ò—Å—Ç–æ—Ä–∏—è –æ–ø–µ—Ä–∞—Ü–∏–π (–∑–∞—â–∏—â–µ–Ω–Ω–∞—è –∫–æ–ø–∏—è)"""
        return self.__transactions.copy()
```

**–ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è:**

```python
print(f"–¢–µ–∫—É—â–∏–π –±–∞–ª–∞–Ω—Å: {account.balance}")

# –ü—Ä–æ—Å–º–æ—Ç—Ä –∏—Å—Ç–æ—Ä–∏–∏

print(f"–ò—Å—Ç–æ—Ä–∏—è –æ–ø–µ—Ä–∞—Ü–∏–π: {history}")

# –û–±—Ä–∞–±–æ—Ç–∫–∞ –æ—à–∏–±–æ–∫

    account.withdraw(5000)  # –û—à–∏–±–∫–∞: –Ω–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ —Å—Ä–µ–¥—Å—Ç–≤
except ValueError as e:
    print(f"–û—à–∏–±–∫–∞: {e}")
```
